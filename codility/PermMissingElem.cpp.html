// PermMissingElem
// Find the missing element in a given permutation. 
#include <vector>
using namespace std;

int PermMissingElem(vector<int> &A) 
{
    size_t sum = A.size() + 1;
    for (size_t i = A.size(); i > 0; --i)
    {
        sum -= A[i - 1];
        sum += i;
    }
    return sum;
}
