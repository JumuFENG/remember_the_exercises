// StoneWall
// Cover "Manhattan skyline" using the minimum number of rectangles.
#include <vector>
#include <stack>
using namespace std;

int StoneWall(vector<int> &H)
{
    if (H.size() <= 1)
    {
        return H.size();
    }

    stack<int> sLst;
    int count = 0;
    for (size_t i = 0; i < H.size(); ++i)
    {
        while (!sLst.empty() && H[i] < sLst.top())
        {
            sLst.pop();
            count++;
        }
        if (sLst.empty() || H[i] > sLst.top())
        {
            sLst.push(H[i]);
        }
    }
    return count + sLst.size();
}
