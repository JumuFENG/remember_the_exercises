// PermCheck
// Check whether array A is a permutation.
#include <vector>
using namespace std;

int PermCheck(vector<int> &A)
{
    vector<int> counts(A.size(), 0);
    for (auto a : A)
    {
        if (size_t(a - 1) >= A.size())
        {
            return 0;
        }
        counts[a - 1] ++;
    }
    for (auto c : counts)
    {
        if (c != 1)
        {
            return 0;
        }
    }
    return 1;
}
