// CountDiv
// Compute number of integers divisible by k in range[a..b].
#include <vector>
using namespace std;

int CountDiv(int A, int B, int K)
{
    if (A < 0 || B < 0 || A > B || K <= 0)
    {
        return 0;
    }

    int a = (A % K == 0) ? A : K * (A / K + 1);
    if (a > B)
    {
        return 0;
    }

    int b = (B % K == 0) ? B : (B / K)*K;
    if (b == a)
    {
        return 1;
    }

    if (b > a)
    {
        return (b - a) / K + 1;
    }
    return 0;
}
