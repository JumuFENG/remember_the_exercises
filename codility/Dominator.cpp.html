// Dominator
// Find an index of an array such that its value occurs at more than half of indices in the array.
#include <vector>
using namespace std;

int Dominator(vector<int>& A)
{
    if (A.size() < 1)
    {
        return -1;
    }

    size_t count = 1;
    int lead = A[0];
    for (size_t i = 1; i < A.size(); ++i)
    {
        if (count == 0)
        {
            lead = A[i];
        }

        if (lead == A[i])
        {
            count++;
        }
        else
        {
            count--;
        }
    }
    
    if (count <= 0)
    {
        return -1;
    }

    count = 0;
    int idx = 0;
    for (size_t i = 0; i < A.size(); ++i)
    {
        if (A[i] == lead)
        {
            count++;
            idx = i;
        }
    }

    if (count + count <= A.size())
    {
        return -1;
    }
    return idx;
}
