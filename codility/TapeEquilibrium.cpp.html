// TapeEquilibrium
// Minimize the value | (A[0] + ... + A[P - 1]) - (A[P] + ... + A[N - 1]) | .
#include <vector>
using namespace std;

int TapeEquilibrium(vector<int> &A)
{
    if (A.size() < 2)
    {
        return 0;
    }

    int sum = 0;
    for (auto a : A)
    {
        sum += a;
    }

    int leftSum = A[0];
    int mVal = abs(sum - A[0] - A[0]);
    for (size_t i = 1; i < A.size() - 1; ++i)
    {
        leftSum += A[i];
        int cur = abs(sum - leftSum - leftSum);
        if (cur < mVal)
        {
            mVal = cur;
        }
    }
    return mVal;
}
