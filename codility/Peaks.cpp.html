// Peaks
// Divide an array into the maximum number of same - sized blocks, each of which should contain an index P such that A[P - 1] < A[P] > A[P + 1].
#include <vector>
using namespace std;

int Peaks(vector<int> A)
{
    if (A.size() <= 2)
    {
        return 0;
    }

    vector<bool> peaks(A.size(), false);
    int cnt = 0;
    for (size_t i = 1; i < A.size() - 1; ++i)
    {
        if (A[i-1] < A[i] && A[i] > A[i+1])
        {
            cnt++;
            peaks[i] = true;
        }
    }

    for (size_t i = cnt; i > 0; --i)
    {
        if (A.size() % i != 0)
        {
            continue;
        }
        bool hasPeak = true;
        for (size_t j = 0; j < i; ++j)
        {
            bool kHasPeak = false;
            for (size_t k = 0; k < A.size() / i; k++)
            {
                if (peaks[j * (A.size() / i) + k])
                {
                    kHasPeak = true;
                    break;
                }
            }
            hasPeak = kHasPeak;
            if (!hasPeak)
            {
                break;
            }
        }
        if (hasPeak)
        {
            return i;
        }
    }
    return 0;
}
