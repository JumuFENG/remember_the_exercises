// MaxProductOfThree
// Maximize A[P] * A[Q] * A[R] for any triplet(P, Q, R).
#include <vector>
#include <algorithm>
using namespace std;

int MaxProductOfThree(vector<int> &A) 
{
    if (A.size() < 3)
    {
        return -1;
    }

    sort(A.begin(), A.end());
    size_t idx = A.size() - 1;
    int r = max(A[0] * A[1] * A[2], A[0] * A[1] * A[idx]);
    return max(A[idx] * A[idx - 1] * A[idx - 2], r);
}
