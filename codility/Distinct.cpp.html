// Distinct
// Compute number of distinct values in an array.
#include <vector>
#include <algorithm>
using namespace std;

int Distinct(vector<int> &A)
{
    if (A.size() <= 1)
    {
        return A.size();
    }

    sort(A.begin(), A.end());
    int count = 1;
    for (size_t i = 1; i < A.size(); ++i)
    {
        if (A[i] != A[i-1])
        {
            count++;
        }
    }
    return count;
}
