// PassingCars
// Count the number of passing cars on the road.
#include <vector>
using namespace std;

int PassingCars(vector<int> &A)
{
    if (A.size() <= 1)
    {
        return 0;
    }

    size_t p = 0;
    for (; p < A.size(); ++p)
    {// skip leading 1s
        if (A[p] == 0)
        {
            break;
        }
    }

    size_t countOne = 0;
    for (size_t q = p + 1; q < A.size(); ++q)
    { // count 1s
        if (A[q] == 1)
        {
            countOne++;
        }
    }

    int count = 0;
    for (; p < A.size(); ++p)
    {
        if (A[p] == 0)
        {
            count += countOne;
        }
        else if (A[p] == 1)
        {
            countOne--;
        }

        if (count > 1000000000)
        {
            return -1;
        }
    }

    return count;
}
