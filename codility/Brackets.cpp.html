// Brackets
// Determine whether a given string of parentheses(multiple types) is properly nested.
#include <stack>
using namespace std;

int Brackets(string &S)
{
    stack<char> brs;
    for (size_t i = 0; i < S.size(); ++i)
    {
        if (S[i] == '{' || S[i] == '}' || S[i] == '[' || S[i] == ']' || S[i] == '(' || S[i] == ')')
        {
            if (brs.empty())
            {
                brs.push(S[i]);
                continue;
            }

            if ((brs.top() == '{' && S[i] == '}')
                || (brs.top() == '[' && S[i] == ']')
                || (brs.top() == '(' && S[i] == ')'))
            {
                brs.pop();
            }
            else
            {
                brs.push(S[i]);
            }
        }
    }
    return brs.size() == 0 ? 1 : 0;
}
