// Triangle
// Determine whether a triangle can be built from a given set of edges.

// 若存在有序升数组A的三个元素能组成三角形的三边，则一定存在相邻的三个数能组成三角形的三边。
// 设存在三个数 a + b > c; a = A[i], b=A[k], c = A[j]; i <= k - 1, j >= k+1;
// 则 a1 = A[k-1] >= A[i] = a; c1 = A[k+1] <= A[j] = c;
// a1 + b > a + b > c > c1;
// 所以 一定有  A[k-1] + A[k] > A[k+1]
#include <vector>
#include <algorithm>
using namespace std;

int Triangle(vector<int> &A)
{
    if (A.size() < 3)
    {
        return 0;
    }

    sort(A.begin(), A.end());
    auto a = A.begin();
    while (*a <= 0 && a != A.end())
    {
        a++;
    }

    if (A.end() - a < 3)
    {
        return 0;
    }

    for (; a != A.end() - 2; ++a)
    {
        if (*(a + 2) - *(a + 1) < *a)
        {
            return 1;
        }
    }
    return 0;
}
