// CountFactors
// Count factors of given number n.
#include <vector>
using namespace std;

int CountFactors(int N) {
    if (N < 1)
    {
        return 0;
    }

    if (N == 1)
    {
        return 1;
    }

    if (N < 4)
    {
        return 2;
    }

    int s = (int)sqrt(N);
    int cnt = 0;
    if (s * s == N)
    {
        cnt++;
    }
    else
    {
        s++;
    }
    cnt += 2;
    for (int i = 2; i < s; ++i)
    {
        if (N % i == 0)
        {
            cnt += 2;
        }
    }
    return cnt;
}
