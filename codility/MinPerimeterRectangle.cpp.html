// MinPerimeterRectangle
// Find the minimal perimeter of any rectangle whose area equals N.
#include <cmath>
using namespace std;

int MinPerimeterRectangle(int N)
{
    if (N < 1)
    {
        return 0;
    }

    if (N == 1)
    {
        return 4;
    }

    if (N < 4)
    {
        return 2 * (1 + N);
    }

    int s = (int)sqrt(N);
    if (s * s == N)
    {
        return 4 * s;
    }

    for (int i = s; i > 0; --i)
    {
        if (N % i == 0)
        {
            return 2 * (i + N / i);
        }
    }
    return 0;
}
